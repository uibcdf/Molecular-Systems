radius = 2**(-5.0/6.0)*sigma

if l_box is None:
    volume_particles = n_particles*(4.0/3.0)*np.pi*radius**3
    volume = volume_particles/reduced_density
    l_box = volume**(1/3)
    print('Side of the box: {}'.format(l_box))
else:
    volume_particles = n_particles*(4.0/3.0)*np.pi*radius**3
